## Install

```bash
npm install --save react-inplace-editor
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-inplace-editor'
import 'react-inplace-editor/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
